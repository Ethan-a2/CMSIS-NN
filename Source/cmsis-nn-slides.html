Activation Function: ReLU

File: ActivationFunctions/arm_relu_q7.c

void arm_relu_q7(int8_t *data, uint16_t size)
{
#if defined(ARM_MATH_DSP) && !defined(ARM_MATH_MVEI)
    /* DSP extension optimization: process 4 elements at once */
    uint16_t i = size >> 2;
    int8_t *input = data;
    int32_t in, buf, mask;
    int8_t *output = data;  // in-place

    while (i)
    {
        /* Read 4 signed bytes */
        in = arm_nn_read_s8x4_ia((const int8_t **)&input);

        /* Extract sign bits and expand to all bytes */
        buf = (int32_t)ROR((uint32_t)in & 0x80808080, 7);

        /* Create mask: 0xFF if negative, 0x00 if positive */
        mask = QSUB8(0x00000000, buf);
        /* Apply mask: keep original if positive, zero if negative */
        arm_nn_write_s8x4_ia(&output, in & (~mask));
        i--;
    }
    /* Handle remainder (0-3 elements) */
    i = size & 0x3;
    while (i)
    {
        if (*input < 0)
            *output = 0;
        else
            *output = *input;
        input++;
        output++;
        i--;
    }
#else
    /* Pure C reference implementation */
    for (uint16_t i = 0; i < size; i++)
        if (data[i] < 0) data[i] = 0;
#endif
}

Key Techniques:

  • SIMD parallelization: 4 elements at once using DSP intrinsics
  • Bit manipulation: Extract sign bit via AND mask, create zero-mask with QSUB8
  • Conditional compile: Auto-select optimal implementation
  • In-place operation: No extra buffer needed

Convolution: im2col + GEMM Pattern

File: ConvolutionFunctions/arm_convolve_s8.c

arm_cmsis_nn_status arm_convolve_s8(
    const cmsis_nn_context *ctx,
    const cmsis_nn_conv_params *conv_params,
    const cmsis_nn_per_channel_quant_params *quant_params,
    const cmsis_nn_dims *input_dims,
    const int8_t *input_data,
    const cmsis_nn_dims *filter_dims,
    const int8_t *filter_data,
    const cmsis_nn_dims *bias_dims,
    const int32_t *bias_data,
    const cmsis_nn_dims *upscale_dims,  // for transposed conv
    const cmsis_nn_dims *output_dims,
    int8_t *output_data)
{
    int16_t *buffer_a = (int16_t *)ctx->buf;  // Accumulation buffer (int16)

    /* Extract dimensions */
    const uint16_t input_x  = input_dims->w;
    const uint16_t input_y  = input_dims->h;
    const uint16_t input_ch = input_dims->c;
    const uint16_t kernel_x = filter_dims->w;
    const uint16_t kernel_y = filter_dims->h;
    const uint16_t kernel_ch = filter_dims->c;  // == input_ch
    const uint16_t output_x  = output_dims->w;
    const uint16_t output_y  = output_dims->h;
    const uint16_t output_ch = output_dims->c;

    /* Convection parameters */
    const uint16_t pad_x = conv_params->padding.w;
    const uint16_t pad_y = conv_params->padding.h;
    const uint16_t stride_x = conv_params->stride.w;
    const uint16_t stride_y = conv_params->stride.h;
    const int32_t input_offset = conv_params->input_offset;
    const int32_t out_offset = conv_params->output_offset;
    const int32_t out_activation_min = conv_params->activation.min;
    const int32_t out_activation_max = conv_params->activation.max;

    /* Special case: Depthwise or Grouped */
    const int32_t groups = input_ch / kernel_ch;
    const int32_t output_ch_per_group = output_ch / groups;

    /* Main loops: batch -> groups -> output positions */
    for (int32_t i_batch = 0; i_batch < input_batches; i_batch++)
    {
        int32_t lhs_rows = 0;  // rows accumulated before GEMM call
        int8_t *im2col_buf;

        for (int32_t i_group = 0; i_group < groups; i_group++)
        {
            int8_t *out_ptr = output_data + i_group * output_ch_per_group;
            for (int32_t i_out_y = 0; i_out_y < output_y; i_out_y++)
            {
                for (int32_t i_out_x = 0; i_out_x < output_x; i_out_x++)
                {
                    /* Build im2col buffer: extract input patch */
                    im2col_buf = im2col_direct(input_data, ...);
                    lhs_rows++;
                    
#if defined(ARM_MATH_MVEI)
                    /* Helium: Process 4 rows at once */
                    if (lhs_rows == 4)
                    {
                        arm_nn_mat_mult_nt_t_s8(
                            buffer_a, filter_data_ptr, bias_data_ptr,
                            out_ptr, output_mult_ptr, output_shift_ptr,
                            lhs_rows, output_ch_per_group, rhs_cols,
                            input_offset, out_offset,
                            out_activation_min, out_activation_max,
                            output_ch, aligned_rhs_cols);
                        out_ptr += lhs_rows * output_ch;
                        lhs_rows = 0;
                    }
#elif defined(ARM_MATH_DSP)
                    /* DSP: Process 2 rows at once */
                    if (lhs_rows == 2)
                    {
                        out_ptr = arm_nn_mat_mult_kernel_s8_s16(...);
                        lhs_rows = 0;
                    }
#endif
                }  // x
            }  // y
            /* Process leftover rows */
        }  // groups
    }  // batch
    return ARM_CMSIS_NN_SUCCESS;
}

Algorithm Strategy

  • im2col conversion: Extract input patches on-the-fly
  • Matrix multiplication: GEMM with accumulated rows
  • Partial accumulation: batch GEMM calls for MVE/DSP
  • Quantization: Per-channel requantization after sum

Optimization Levels

  • MVE (Helium): 4 rows × 128-bit vectors = 64 int8 multiplies/cycle
  • DSP: 2 rows × 32-bit SIMD = 8 int8 multiplies/cycle
  • Reference: Scalar accumulation

Fully Connected Layer

File: FullyConnectedFunctions/arm_fully_connected_s8.c

arm_cmsis_nn_status arm_fully_connected_s8(
    const cmsis_nn_context *ctx,
    const cmsis_nn_fc_params *fc_params,
    const cmsis_nn_per_tensor_quant_params *quant_params,
    const cmsis_nn_dims *input_dims,
    const int8_t *input,
    const cmsis_nn_dims *filter_dims,
    const int8_t *kernel,
    const cmsis_nn_dims *bias_dims,
    const int32_t *bias,
    const cmsis_nn_dims *output_dims,
    int8_t *output)
{
    int32_t batch_cnt = input_dims->n;
    const int32_t input_depth = filter_dims->n;
    const int32_t output_depth = filter_dims->c;

    /* Precompute kernel sums for requantization */
    const int32_t *kernel_sum = (const int32_t *)ctx->buf;
    arm_nn_mat_mult_nt_t_s8_gen_kernel_sum(
        kernel, kernel_sum,
        output_depth, input_depth,
        fc_params->filter_offset);

    while (batch_cnt)
    {
        arm_nn_vec_mat_mult_t_s8(
            input,           // (1 × C_in)
            kernel,          // (C_in × C_out)
            kernel_sum,      // [C_out] precomputed sums
            bias,            // [C_out] or NULL
            output,          // (1 × C_out)
            fc_params->input_offset,
            fc_params->output_offset,
            quant_params->multiplier,
            quant_params->shift,
            input_depth,
            output_depth,
            fc_params->activation.min,
            fc_params->activation.max,
            1,                                 // Process 1 row at a time
            fc_params->filter_offset);        // Per-tensor filter offset

        input += input_depth;
        output += output_depth;
        batch_cnt--;
    }
    return ARM_CMSIS_NN_SUCCESS;
}

Key Differences from Convolution:

  • No spatial dimensions: Fully connected is just vector-matrix multiply
  • Simpler loop structure: Only batch loop, no spatial loops
  • Kernel sum precomputation: Used for requantization optimization
  • Delegates to: arm_nn_vec_mat_mult_t_s8 in SupportFunctions

/* Compiler-specific force inlining */
#define __STATIC_INLINE static inline
#define __STATIC_FORCEINLINE __attribute__((always_inline)) static inline

/* Data packing macro */
#define PACK_S8x4_32x1(v0,v1,v2,v3) \
    ((((int32_t)(v0) << 0) & 0xFF) | \
     (((int32_t)(v1) << 8) & 0xFF00) | \
     (((int32_t)(v2) << 16) & 0xFF0000) | \
     (((int32_t)(v3) << 24) & 0xFF000000))

/* Read/write helpers for SIMD access */
__STATIC_FORCEINLINE int32_t arm_nn_read_s8x4_ia(const int8_t **in)
{
    int32_t val = (int32_t)*((const int8_t *)*in);
    *in += 4;
    return val;
}

/* Check if fast path applicable */
__STATIC_INLINE bool arm_nn_is_convolve_1x1(
    const cmsis_nn_conv_params *conv_params,
    const cmsis_nn_dims *input_dims,
    const cmsis_nn_dims *filter_dims)
{
    return (conv_params->padding.w == 0) &&
           (conv_params->padding.h == 0) &&
           (filter_dims->w == 1) && (filter_dims->h == 1) &&
           (conv_params->dilation.w == 1) && (conv_params->dilation.h == 1);
}

API Pattern: Buffer Management

Functions requiring scratch memory use a two-call pattern:

/* Step 1: Query buffer size */
int32_t arm_convolve_s8_get_buffer_size(
    const cmsis_nn_dims *input_dims,
    const cmsis_nn_dims *filter_dims);

/* Step 2: Allocate and pass context */
cmsis_nn_context ctx;
ctx.buf = malloc(buf_size);
ctx.size = buf_size;  // or 0, kernel manages

arm_cmsis_nn_status status = arm_convolve_s8(&ctx, ...);

/* Step 3: Security cleanup - caller responsible */
memset(ctx.buf, 0, buf_size);  // Clear sensitive data
free(ctx.buf);

Buffer Size Estimation

/* Example: Convolution buffer calculation */
int32_t arm_convolve_s8_get_buffer_size(
    const cmsis_nn_dims *input_dims,
    const cmsis_nn_dims *filter_dims)
{
    int32_t im2col_size = 0;

    /* Calculate temporary im2col buffer */
    if (arm_nn_is_convolve_1x1(...))
        im2col_size = input_dims->h * input_dims->w * filter_dims->c *
                      sizeof(int8_t);  // Reuse buffer for original data
    else
        im2col_size = input_dims->n * input_dims->c * input_dims->h * input_dims->w *
                      sizeof(int8_t);  // Full im2col

    return im2col_size;
}

Security Considerations:

  • Caller must clear buffer after use to prevent data leakage
  • Buffer size is dynamic, caller can implement custom allocator (pool, stack, etc.)
  • Some functions need no buffer (e.g., ReLU), pass ctx.buf = NULL

Quantization Model

CMSIS-NN uses affine quantization: q = round(x / scale) + zero_point

Per-Tensor Quantization

typedef struct {
    int32_t multiplier;
    int32_t shift;
} cmsis_nn_per_tensor_quant_params;

Single scale for entire tensor. Simple, less accurate.

Requantization Formula:

/* After dot product accumulation (int32 result) */
int32_t requantized = ((acc * multiplier) >> shift) + output_offset;
requantized = CLAMP(requantized, activation_min, activation_max);
output[i] = (int8_t)requantized;

Per-Channel Quantization

typedef struct {
    int32_t *multiplier;  // array[output_depth]
    int32_t *shift;       // array[output_depth]
} cmsis_nn_per_channel_quant_params;

Each output channel gets its own scale. More accurate, supports fine calibration.

Usage:

for (int c = 0; c < output_depth; c++) {
    int32_t mult = multiplier[c];
    int32_t shft = shift[c];
    int32_t val = ((acc[c] * mult) >> shft) + output_offset;
    output[c] = CLAMP(val, min, max);
}

Calculating Multiplier & Shift:

Given scale = (input_scale × kernel_scale) / output_scale, compute:

  • real_multiplier = input_scale × kernel_scale / output_scale
  • multiplier = round(real_multiplier × 2^31)
  • shift = 31 - floor(log2(multiplier)) (shift right by 31-bits)

Tooling: TFLite Micro's hybrid_quantization_logger.cc computes these during conversion.

Key Takeaways

1. Layered Architecture High-level wrappers auto-select optimal kernel based on CPU capabilities (MVE > DSP > C), enabling single codebase for all Cortex-M devices.
2. Quantization-First Design Native int8/int16 operations with per-tensor or per-channel scaling. Optimized for TFLite Micro's quantized models.
3. Memory Efficiency Scratch buffer management through context pattern; partial im2col strategy reduces memory footprint compared to naive GEMM.
4. Performance Patterns MVE: 4-accumulator batching, DSP: 2-accumulator; im2col/GEMM approach with optimizations for 1×1 convs and depthwise convs.
5. Portability Conditional compilation abstracts SIMD; reference C fallback ensures correctness on any ARM processor.
6. Security Caller-controlled scratch buffer with required zeroing after use; no static buffers, suitable for secure enclaves.
