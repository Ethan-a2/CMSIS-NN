激活函数实现

激活函数为神经网络引入非线性，CMSIS-NN 提供高效的定点实现：

/*
 * Q7 ReLU 函数
 * 使用 DSP 扩展加速处理
 */
void arm_relu_q7(int8_t *data, uint16_t size)
{
#if defined(ARM_MATH_DSP) && !defined(ARM_MATH_MVEI)
    /* M 核心 DSP 扩展代码 */
    uint16_t i = size >> 2;
    int8_t *input = data;
    int8_t *output = data;
    int32_t in, buf, mask;

    while (i)
    {
        in = arm_nn_read_s8x4_ia((const int8_t **)&input);
        /* 提取最高位 */
        buf = (int32_t)ROR((uint32_t)in & 0x80808080, 7);
        /* 如果 MSB=1, mask 为 0xFF */
        mask = QSUB8(0x00000000, buf);
        arm_nn_write_s8x4_ia(&output, in & (~mask));
        i--;
    }
#else
    /* 无 DSP 扩展的参考实现 */
    for (uint16_t i = 0; i < size; i++)
    {
        if (data[i] < 0)
            data[i] = 0;
    }
#endif
}

static void scale_q31_to_q7_and_clamp(
    const int32_t *buffer,
    int8_t *target,
    int32_t length,
    const int32_t count,
    const int act_min,
    const int act_max)
{
    const int32_t half_count = count / 2;
    for (int i = 0; i < length; i++)
    {
        int32_t sum = buffer[i] > 0 ? 
            (buffer[i] + half_count) : (buffer[i] - half_count);
        sum = sum / count;
        sum = MAX(sum, act_min);
        sum = MIN(sum, act_max);
        target[i] = (int8_t)sum;
    }
}

Softmax 实现

Softmax 将 logits 转换为概率分布，使用数值稳定的近似算法：

/**
 * S8 Softmax 函数
 * 使用 MVEI 向量指令加速指数计算
 */
#if defined(ARM_MATH_MVEI) && !defined(ARM_GCC_12_2_ICE)
static int32x4_t arm_exp_on_negative_values_mve_32x4(
    int32x4_t val)
{
    #define SHIFT_START (24)
    int32_t shift = SHIFT_START;
    int32x4_t mask;

    const int32x4_t val_mod_minus_quarter =
        vandq_s32(val, vdupq_n_s32((1 << SHIFT_START) - 1)) - 
        vdupq_n_s32(1 << SHIFT_START);
    const int32x4_t remainder = vsubq_s32(val_mod_minus_quarter, val);
    const int32x4_t x = vaddq_n_s32(val_mod_minus_quarter << 5, 1 << 28);
    const int32x4_t x2 = MUL_SAT_MVE(x, x);
    // 泰勒级数近似 exp(x)
    const int32x4_t op_1 = DIV_POW2_MVE(MUL_SAT_MVE(x2, x2), 2) + 
        MUL_SAT_MVE(x2, x);
    int32x4_t result = vdupq_n_s32(1895147668) + 
        MUL_SAT_MVE(vdupq_n_s32(1895147668), op_2);
    return result;
}
#endif

量化策略

CMSIS-NN 支持多种量化格式，优化内存和计算效率：

量化类型 位宽 范围 应用场景
S4 4-bit [-8, 7] 极致压缩
S8 8-bit [-128, 127] 主流部署
S16 16-bit [-32768, 32767] 高精度需求
U8 8-bit [0, 255] 无符号激活

硬件优化

CMSIS-NN 针对 Arm Cortex-M 处理器进行深度优化：

DSP 扩展 (SIMD)：使用 SMLAD, SMUAD 等指令一次处理 2-4 个数据
MVEI 扩展：M-Profile Vector Extension 提供 128-bit 向量运算
循环展开：减少分支预测失败，提高指令级并行
内存访问优化：使用 __SXTAB16, __UXTAB16 等指令

使用示例

典型的卷积神经网络推理流程：

// 1. 定义网络参数
cmsis_nn_dims input_dims = {1, 28, 28, 1};
cmsis_nn_dims filter_dims = {3, 3, 1, 16};
cmsis_nn_dims output_dims = {1, 26, 26, 16};

// 2. 配置卷积参数
cmsis_nn_conv_params conv_params = {
    .padding = {0, 0},
    .stride = {1, 1},
    .dilation = {1, 1},
    .input_offset = 128,
    .output_offset = -128,
    .activation = {-128, 127}
};

// 3. 执行卷积
arm_convolve_s8(
    &ctx,
    &conv_params,
    &quant_params,
    &input_dims,
    input_data,
    &filter_dims,
    filter_data,
    &bias_dims,
    bias_data,
    NULL,
    &output_dims,
    output_data
);

// 4. 应用激活函数
arm_relu_q7(output_data, output_size);

// 5. 池化
arm_max_pool_s8(
    &ctx,
    &pool_params,
    &output_dims,
    output_data,
    &filter_dims,
    &pooled_dims,
    pooled_data
);
